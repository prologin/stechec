\subsection{Fonctions d'information}

\textbf{BAD\_ARGUMENT} ou \textbf{NO\_UNIT} quand au moins un des
arguments est incorrect (même si cela n'est pas précisé pour chaque
fonction). Par exemple, cela se produit si vous appelez
la fonction \texttt{regarde} avec \texttt{x=-1} et \texttt{y=5142}.

int deplace_leucocyte(int x, int y)
\end{lst-c++}

\noindent
\begin{tabular}[t]{@{\extracolsep{0pt}}>{\bfseries}lp{10cm}}
Description~: & Deplace un de ses leucocyte \emph{vers} la position indiquee. Il ne se deplace que
d'une seule case par tour.
 \\


Paramètres~: &
\begin{tabular}[t]{@{\extracolsep{0pt}}ll}



        \textsl{x}~: & Coordonnee x à atteindre. \\



        \textsl{y}~: & Coordonnee y à atteindre. \\


  \end{tabular} \\



Retour~: & SUCCESS si le joueur peut se déplacer à la fin du tour (et il le fera),
ou INVALID\_COMMAND ou BAD\_ARGUMENT en cas d'impossibilité.
 \\




\end{tabular} \\[0.3cm]
\end{minipage}

\functitle{phagocyte}

\begin{lst-c++}
int phagocyte(int x, int y)
\end{lst-c++}

\noindent
\begin{tabular}[t]{@{\extracolsep{0pt}}>{\bfseries}lp{10cm}}
Description~: & Phagocytose de l'entité qui se trouve aux coordonnées indiquées.
S'il s'agit d'un virus dont le type n'est pas encore connu,
l'appel à cette fonction immobilisera plus longtemps le leucocyte
que si le type est connu (apprentissage de la signature).
 \\


Paramètres~: &
\begin{tabular}[t]{@{\extracolsep{0pt}}ll}



        \textsl{x}~: & Coordonnée x de l'entité. \\



        \textsl{y}~: & Coordonnée y de l'entité. \\


  \end{tabular} \\



Retour~: & SUCCESS si le leucocyte réussi à phagocyter
BAD\_ARGUMENT ou INVALID\_COMMAND sinon
 \\




\end{tabular} \\[0.3cm]
\end{minipage}
