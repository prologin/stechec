\section{Le projet TowBowlTactics}

\frame
{
  \begin{block}{Présentation de Blood Bowl}
    \begin{itemize}
      \item Jeu de plateau édité par Games Worshop depuis 1986.
      \item Figurines, différentes races.
      \item Allie convivialité, rapidité, hasard et stratégie.
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Tow Bowl tactics}
    \begin{itemize}
      \item Adaptation vidéo de Blood Bowl.
      \item Née de la volontée d'une personne, Pascal Bourut, en 2001.
      \item Programmé en C++/SDL, réalisé en 1 an.
      \item Communautée se formant autour du jeu, mais pas
        suffisamment de programmeur pour le faire vivre.
      \item Problèmes :
        \begin{itemize}
          \item Code trop peu documenté et monolithique.
          \item Unique développeur manquant de temps pour continuer
            le développement et la maintenance du projet.
        \end{itemize}
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Légalement}
    \begin{itemize}
      \item Projet entièrement en GPL.
      \item Droits d'adaptation vis-à-vis de GW flou.
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Reprise fin 2005}
    \begin{itemize}
      \item Reprise amorcée par un fan: Mathieu.
      \item Soutient technique de Nekeme prod.
      \item Mise en place rapide d'une solide infrastructure:
        \begin{itemize}
          \item site web,
          \item forum,
          \item wiki,
          \item subversion,
          \item listes de diffusion.
        \end{itemize}
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Recrutement et organisation de l'équipe}
    \begin{itemize}
      \item Noyau de 3 personnes mais pas d'organisation formelle.
      \item Proposition d'aide constante d'une communauté toujours active.
      \item Élaboration d'une proposition technique qui servira de base aux
        discutions.
    \end{itemize}
  \end{block}
}

\section{Techniquement}


\frame
{
  \begin{block}{Choix technologique}
    \begin{itemize}
      \item C++/SDL, XML pour la configuration et s'interfacer avec
        d'autres projets.
      \item Doxygen et \LaTeX pour la documentation.
      \item Internationalisation prévue.
      \item Le minimum de dépendances possible.
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Compatibilité}
    \begin{itemize}
      \item SDL est portable.
      \item Choix du support de VC++ autant que possible.
    \end{itemize}
  \end{block}
}

\frame
{
  \begin{block}{Modélisation}
    \begin{itemize}
      \item Système modulaire, client/serveur distinct.
        \begin{itemize}
          \item Serveur: gestion de l'arbitrage et transfert des données.
          \item Client: gestion des joueurs et des données.
        \end{itemize}
      \item Connexion TCP/IP entre le client et le serveur.
      \item Plusieurs interface possible, console, 2D, \ldots
    \end{itemize}
  \end{block}  
}
